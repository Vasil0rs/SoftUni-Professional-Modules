import java.lang.reflect.Array;
import java.util.Arrays;
import java.util.Scanner;
import java.util.function.Function;

public class SumNumbers {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int[] nums = Arrays.stream(scanner.nextLine().split(", "))
                .mapToInt(Integer::parseInt)
                .toArray();

        Function<int[], String> getCountString = arr -> String.format("Count = %d", arr.length);

        System.out.println(getCountString.apply(nums));

        Function<int[], String> getSumString = arr -> {
            return "Sum = " + Arrays.stream(arr).sum();
        };
        System.out.println(getSumString.apply(nums));
    }
}
