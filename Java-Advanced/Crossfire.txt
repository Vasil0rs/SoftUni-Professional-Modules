import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Crossfire {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        String[] dimensions = scanner.nextLine().split("\\s+");

        int rows = Integer.parseInt(dimensions[0]);
        int cols = Integer.parseInt(dimensions[1]);

        List<List<Integer>> listMatrix = new ArrayList<>();

        int count = 1;

        for (int row = 0; row < rows; row++) {
            listMatrix.add(new ArrayList<>());

            for (int col = 0; col < cols; col++) {
                listMatrix.get(row).add(count++);
            }
        }

        String input = scanner.nextLine();

        while (!input.equals("Nuke it from orbit")){
            String[] data = input.split("\\s+");

            int row = Integer.parseInt(data[0]);
            int col = Integer.parseInt(data[1]);
            int radius = Integer.parseInt(data[2]);

            for (int i = row - radius; i <= row + radius; i++) {
                if (isValid(i, col, listMatrix) && i != row){
                    listMatrix.get(i).remove(col);
                }
            }

            for (int i = col + radius; i >= col - radius; i--) {
                if (isValid(row, i, listMatrix)){
                    listMatrix.get(row).remove(i);
                }
            }
            listMatrix.removeIf(List::isEmpty);

            input = scanner.nextLine();
        }

        for(List<Integer> numbers : listMatrix){
            for (Integer number : numbers) {
                System.out.print(number + " ");
            }
            System.out.println();
        }
    }

    private static boolean isValid(int row, int col, List<List<Integer>> matrix) {
        return row >= 0 && row < matrix.size() && col >= 0 && col < matrix.get(row).size();
    }
}
